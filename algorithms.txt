Roots of quadratic equation:
start
var a, b, c, d, r1, r2 as real
input a, b, c
d = b*b - 4*a*c
if d >= 0 then
    r1 = (-b + sqrt(d)) / (2*a)
    r2 = (-b - sqrt(d)) / (2*a)
    print r1, r2
else
    print "Imaginary roots"
end if
end

check if the first number is multiple of the second:
start
var x, y as integer
input x, y
if x mod y = 0 then
    print "x is a multiple of y"
else
    print "x is not a multiple of y"
end if
end
