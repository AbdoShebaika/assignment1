Weekday using switch:
start
var day as integer
input day
switch(day)
    case 1: print "Monday"
    case 2: print "Tuesday"
    case 3: print "Wednesday"
    case 4: print "Thursday"
    case 5: print "Friday"
    case 6: print "Saturday"
    case 7: print "Sunday"
    default: print "Invalid day"
end switch
end




Calcuater using switch:
start
var a, b, result as real
var op as char
input a, op, b
switch(op)
    case '+': result = a + b
    case '-': result = a - b
    case '*': result = a * b
    case '/': result = a / b
    default: print "Invalid operator"
end switch
print result
end

Food order menu:
start
var type as char
var quantity, price, total as integer
input type, quantity
switch(type)
    case 'B': price = 200
    case 'F': price = 50
    case 'P': price = 500
    case 'S': price = 150
    default: print "Invalid food"
end switch
total = price * quantity
print total
end




Digit to word:
start
